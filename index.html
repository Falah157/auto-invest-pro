app.use(express.json()); app.use(cors()); // Helper: sign query string for Binance function sign(query, secret) { return crypto.createHmac('sha256', secret).update(query).digest('hex'); } // POST /connect-binance // body: { apiKey, apiSecret } app.post('/connect-binance', async (req, res) => { const { apiKey, apiSecret } = req.body; if (!apiKey || !apiSecret) return res.json({ success:false, error:'Missing credentials' }); try { const timestamp = Date.now(); const q = `timestamp=${timestamp}`; const signature = sign(q, apiSecret); const response = await axios.get(`https://api.binance.com/api/v3/account?${q}&signature=${signature}`, { headers: { 'X-MBX-APIKEY': apiKey }, timeout: 10000 }); return res.json({ success:true, data: { account: response.data } }); } catch (e) { return res.json({ success:false, error: e.response?.data || e.message }); } }); // POST /market-data // body: { apiKey, apiSecret, symbols: ['BTCUSDT', ...] } app.post('/market-data', async (req, res) => { const symbols = req.body.symbols || [ 'BTCUSDT','ETHUSDT','BNBUSDT','SOLUSDT','XRPUSDT','ADAUSDT','AVAXUSDT','DOTUSDT','DOGEUSDT','MATICUSDT' ]; try { const calls = symbols.map(s => axios.get(`https://api.binance.com/api/v3/ticker/24hr?symbol=${s}`)); const results = await Promise.allSettled(calls); const data = results.map((r, i) => { if (r.status === 'fulfilled') return r.value.data; return { symbol: symbols[i], lastPrice: null, priceChangePercent: null }; }); return res.json({ success:true, data }); } catch (e) { return res.json({ success:false, error: e.message }); } }); // POST /order // body: { apiKey, apiSecret, symbol, quoteOrderQty } app.post('/order', async (req, res) => { const { apiKey, apiSecret, symbol, quoteOrderQty } = req.body; if (!apiKey || !apiSecret || !symbol || !quoteOrderQty) return res.json({ success:false, error:'Missing params' }); try { const timestamp = Date.now(); // Use quoteOrderQty for market buy by quote asset const q = `symbol=${symbol}USDT&side=BUY&type=MARKET"eOrderQty=${encodeURIComponent(quoteOrderQty)}×tamp=${timestamp}`; const signature = sign(q, apiSecret); const url = `https://api.binance.com/api/v3/order?${q}&signature=${signature}`; const response = await axios.post(url, {}, { headers: { 'X-MBX-APIKEY': apiKey }, timeout: 15000 }); return res.json({ success:true, data: response.data }); } catch (e) { return res.json({ success:false, error: e.response?.data || e.message }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log('Server listening on', PORT)); ``` > **Important:** the `server.js` above uses **public Binance mainnet** endpoints. For testing use Binance testnet endpoints (see notes below). --- ## How to run (quick) 1. Create a new folder and put `index.html` and `server.js` (or use the code blocks above). 2. `cd` into folder and `npm init -y` 3. `npm i express axios crypto cors` 4. `node server.js` 5. Make sure your backend is reachable over HTTPS. If testing locally, use `ngrok http 3000` and use the `https` ngrok URL in `index.html` as Server URL. 6. Enter API Key & Secret in the frontend; it will be sent to your backend which will call Binance. --- ## Security & production notes (read!) * **Never** store users' raw API secrets in plain text. This example forwards them for immediate use only. In production you should store encrypted secrets or use user OAuth/Connect flows when available. * Use HTTPS and host the server in a trusted environment (Heroku, Render, Vercel serverless function, Replit with paid HTTPS, or your own VPS with TLS). * For safety, implement rate-limiting, input validation, and smaller scopes (API key with only trading rights you need). * Consider using Binance sub-accounts or API keys with IP whitelist. * For automated scheduled orders use a secure worker process that reads encrypted credentials, not the browser. --- ## Testnet usage (if you want to use Binance testnet) * Replace `https://api.binance.com` with `https://testnet.binance.vision` for REST endpoints. * Testnet sometimes uses different endpoints; confirm in Binance docs. --- ## Next steps I can do for you (choose any): * Provide a version that stores API keys encrypted in a database (and how to decrypt for scheduled jobs) * Make this work with Binance testnet and show example requests/responses * Deploy this backend to Replit or Render and give you the live `SERVER_URL` * Convert backend to serverless functions (Vercel / Netlify / Cloud Functions) --- If you want, tell me which option to do next and I will prepare it. *â€” End of fixed code bundle â€”*
